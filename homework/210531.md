package com.may31.homework;

import java.awt.Color;
import java.awt.GridLayout;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;
import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JOptionPane;
import javax.swing.JPanel;

public class Homework01 extends JFrame implements KeyListener {
	private static final int ROAD = 0;
	private static final int WALL = 1;
	private static final int START = 2;
	private static final int END = 3;
	private static final int CURRENT = 4;
		
	private static final int ROW = 8;
	private static final int COL = 8;
		
	private static final Color[] COLOR = { 
			new Color(250, 237, 239),  // ROAD 의 색상 0
			new Color(33, 32, 32),       // WALL 의 색상 1
			new Color(235, 52, 82),    // START 의 색상 2
			new Color(74, 52, 237),    // END 의 색상 3
			new Color(207, 52, 235)    // CURRENT 의 색상 4
			};
		
	private static final int[][] MAP = { 
		{ START, ROAD, WALL, WALL, WALL, ROAD, WALL, ROAD },
		{ ROAD, ROAD, ROAD, ROAD, WALL, ROAD, WALL, ROAD }, 
		{ WALL, WALL, WALL, ROAD, ROAD, ROAD, ROAD, ROAD },
		{ WALL, ROAD, ROAD, ROAD, WALL, ROAD, WALL, ROAD }, 
		{ WALL, WALL, WALL, ROAD, WALL, WALL, WALL, ROAD },
		{ ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, WALL, END },
		{ ROAD, WALL, WALL, WALL, WALL, WALL, WALL, ROAD },
		{ ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, ROAD }, 
		};
		   
		   
	class button {
		
		JButton btn;
		int func;
		
		button(int f) {
			this.btn = new JButton();
			this.func = f;
		}
	}
		   
	private JPanel wholePanel;
	private button[][] buttonMap = new button[ROW][COL];
	private int currentRow;  //현재 위치
	private int currentCol;  //현재 위치
	
	private void setToOriColor(int r, int c) {
		buttonMap[r][c].btn.setBackground(COLOR[buttonMap[r][c].func]);
		return;
	}
	
	private void setCurrentColor(int r, int c) {
		//current이면 CURRENT색으로
		buttonMap[r][c].btn.setBackground(COLOR[CURRENT]);
		return;
	}
	
	@Override
	public void keyTyped(KeyEvent e) {}
	
	@Override
	public void keyReleased(KeyEvent e) {}
	
	@Override
	public void keyPressed(KeyEvent e) {
		
		int key = e.getKeyCode();
		
		switch(key) {
			case KeyEvent.VK_RIGHT:
				if (currentCol == 7 || buttonMap[currentRow][currentCol + 1].func == WALL) {
					break;
				}
				setToOriColor(currentRow, currentCol);
				currentCol++;
				setCurrentColor(currentRow, currentCol);
				break;
			case KeyEvent.VK_LEFT:
				if (currentCol == 0 || buttonMap[currentRow][currentCol - 1].func == WALL) {
					break;
				}
				setToOriColor(currentRow, currentCol);
				currentCol--;
				setCurrentColor(currentRow, currentCol);
				break;
			case KeyEvent.VK_UP:
				if (currentRow == 0 || buttonMap[currentRow - 1][currentCol].func == WALL) {
					break;
				}
				setToOriColor(currentRow, currentCol);
				currentRow--;
				setCurrentColor(currentRow, currentCol);
				break;
			case KeyEvent.VK_DOWN:
				if (currentRow == 7 || buttonMap[currentRow + 1][currentCol].func == WALL) {
					break;
				}
				setToOriColor(currentRow, currentCol);
				currentRow++;
				setCurrentColor(currentRow, currentCol);
				break;
			default:
		}
		
		//승리했을 때
		if (currentRow == 5 && currentCol == 7) {  //END의 위치
			JOptionPane.showMessageDialog(wholePanel, "==WIN==\n축하드립니다.");
			dispose();
		}
	}


	public Homework01() {
		//창 설정
		super("미로찾기");  //창 제목
		setDefaultCloseOperation(EXIT_ON_CLOSE); //창 닫히면 프로그램 종료
		setSize(500, 500);
		setLocationRelativeTo(null);  //화면 중앙에 창 띄우기
		setVisible(true); //창 띄우기
		
		//패널 설정
		wholePanel = new JPanel();
		wholePanel.setLayout(new GridLayout(ROW, COL));
		this.add(wholePanel);  //패널을 프레임에 붙이기
		
		//버튼 설정
		for (int r=0; r<ROW; ++r) {
			for (int c=0; c<COL; ++c) {
				buttonMap[r][c] = new button(MAP[r][c]);
			}
		}
		for (button[] b : buttonMap) {
			for (button c : b) {
				wholePanel.add(c.btn);  //버튼을 패널에 붙이기
				c.btn.addKeyListener(this);  //KeyListener랑 연결
				c.btn.setBackground(COLOR[c.func]);  //색깔 설정
			}
		}
		
		//게임 시작
		currentRow = 0;
		currentCol = 0;
		setCurrentColor(currentRow, currentCol);
   }
   
   public static void main(String[] args) {
	   new Homework01();
   }
}
